28  // Return a reasonable mime type based on the extension of a file.
29  inline beast::string_view mime_type(beast::string_view path)
30  {
31  using beast::iequals;
32  auto const ext = [&path]
33  {
34  auto const pos = path.rfind(".");
35  if(pos == beast::string_view::npos)
36  return beast::string_view{};
37  return path.substr(pos);
38  }();
39  if(iequals(ext, ".htm")) return "text/html";
40  if(iequals(ext, ".html")) return "text/html";
41  if(iequals(ext, ".php")) return "text/html";
42  if(iequals(ext, ".css")) return "text/css";
43  if(iequals(ext, ".txt")) return "text/plain";
44  if(iequals(ext, ".js")) return "application/javascript";
45  if(iequals(ext, ".json")) return "application/json";
46  if(iequals(ext, ".xml")) return "application/xml";
47  if(iequals(ext, ".swf")) return "application/x-shockwave-flash";
48  if(iequals(ext, ".flv")) return "video/x-flv";
49  if(iequals(ext, ".png")) return "image/png";
50  if(iequals(ext, ".jpe")) return "image/jpeg";
51  if(iequals(ext, ".jpeg")) return "image/jpeg";
52  if(iequals(ext, ".jpg")) return "image/jpeg";
53  if(iequals(ext, ".gif")) return "image/gif";
54  if(iequals(ext, ".bmp")) return "image/bmp";

56  if(iequals(ext, ".tiff")) return "image/tiff";

58  if(iequals(ext, ".svg")) return "image/svg+xml";

63  // Append an HTTP rel-path to a local filesystem path.
64  // The returned path is normalized for the platform.
65  inline std::string path_cat(
66  beast::string_view base,
67  beast::string_view path)
68  {
69  if(base.empty())
70  return std::string(path);
71  std::string result(base);
72  #ifdef BOOST_MSVC
73  char constexpr path_separator = '\\';
74  if(result.back() == path_separator)
75  result.resize(result.size() - 1);
76  result.append(path.data(), path.size());
77  for(auto& c : result)
78  if(c == '/')
79  c = path_separator;
80  #else
81  char constexpr path_separator = '/';
82  if(result.back() == path_separator)
83  result.resize(result.size() - 1);
84  result.append(path.data(), path.size());
85  #endif
86  return result;
87  }

90  //
91  // The concrete type of the response message (which depends on the
92  // request), is type-erased in message_generator.
93  template <class Body, class Allocator>
94  http::message_generator handle_request(
95  beast::string_view doc_root,
96  http::request<Body, http::basic_fields<Allocator>>&& req)
97  {
98  // Returns a bad request response
99  auto const bad_request =
100  [&req](beast::string_view why)
101  {
102  http::response<http::string_body> res{http::status::bad_request, req.version()};
103  res.set(http::field::server, BOOST_BEAST_VERSION_STRING);
104  res.set(http::field::content_type, "text/html");
105  res.keep_alive(req.keep_alive());
106  res.body() = std::string(why);
107  res.prepare_payload();
108  return res;
109  };
110 
111  // Returns a not found response
112  auto const not_found =
113  [&req](beast::string_view target)
114  {
115  http::response<http::string_body> res{http::status::not_found, req.version()};
116  res.set(http::field::server, BOOST_BEAST_VERSION_STRING);
117  res.set(http::field::content_type, "text/html");
118  res.keep_alive(req.keep_alive());
119  res.body() = "The resource '" + std::string(target) + "' was not found.";
120  res.prepare_payload();
121  return res;
122  };
123 
124  // Returns a server error response
125  auto const server_error =
126  [&req](beast::string_view what)
127  {
128  http::response<http::string_body> res{http::status::internal_server_error, req.version()};
129  res.set(http::field::server, BOOST_BEAST_VERSION_STRING);
130  res.set(http::field::content_type, "text/html");
131  res.keep_alive(req.keep_alive());
132  res.body() = "An error occurred: '" + std::string(what) + "'";
133  res.prepare_payload();
134  return res;
135  };
136 
137  // Make sure we can handle the method
138  if( req.method() != http::verb::get &&
139  req.method() != http::verb::head)
140  return bad_request("Unknown HTTP-method");
141 
142  // Request path must be absolute and not contain "..".
143  if( req.target().empty() ||
144  req.target()[0] != '/' ||
145  req.target().find("..") != beast::string_view::npos)
146  return bad_request("Illegal request-target");
147 
148  // Build the path to the requested file
149  std::string path = path_cat(doc_root, req.target());
150  if(req.target().back() == '/')
151  path.append("index.html");
152 
153  // Attempt to open the file
154  beast::error_code ec;
155  http::file_body::value_type body;
156  body.open(path.c_str(), beast::file_mode::scan, ec);
157 
158  // Handle the case where the file doesn't exist
159  if(ec == beast::errc::no_such_file_or_directory)
160  return not_found(req.target());
161 
162  // Handle an unknown error
163  if(ec)
164  return server_error(ec.message());
165 
166  // Cache the size since we need it after the move
167  auto const size = body.size();
168 
169  // Respond to HEAD request
170  if(req.method() == http::verb::head)
171  {
172  http::response<http::empty_body> res{http::status::ok, req.version()};
173  res.set(http::field::server, BOOST_BEAST_VERSION_STRING);
174  res.set(http::field::content_type, mime_type(path));
175  res.content_length(size);
176  res.keep_alive(req.keep_alive());
177  return res;
178  }
179 
180  // Respond to GET request
181  http::response<http::file_body> res{
182  std::piecewise_construct,
183  std::make_tuple(std::move(body)),
184  std::make_tuple(http::status::ok, req.version())};
185  res.set(http::field::server, BOOST_BEAST_VERSION_STRING);
186  res.set(http::field::content_type, mime_type(path));
187  res.content_length(size);
188  res.keep_alive(req.keep_alive());
189  return res;
190  }

195  // Handles an HTTP server connection
196  inline net::awaitable<void> do_session(
197  tcp_stream stream,
198  std::shared_ptr<std::string const> doc_root)
199  {
200  // This buffer is required to persist across reads
201  beast::flat_buffer buffer;
202 
203  // This lambda is used to send messages
204  try
205  {
206  for(;;)
207  {
208  // Set the timeout.
209  stream.expires_after(std::chrono::seconds(30));
210 
211  // Read a request
212  http::request<http::string_body> req;
213  co_await http::async_read(stream, buffer, req);
214 
215  // Handle the request
216  http::message_generator msg =
217  handle_request(*doc_root, std::move(req));
218 
219  // Determine if we should close the connection
220  bool keep_alive = msg.keep_alive();
221 
222  // Send the response
223  co_await beast::async_write(stream, std::move(msg), net::use_awaitable);
224 
225  if(! keep_alive)
226  {
227  // This means we should close the connection, usually because
228  // the response indicated the "Connection: close" semantic.
229  break;
230  }
231  }
232  }
233  catch (boost::system::system_error & se)
234  {
235  if (se.code() != http::error::end_of_stream )
236  throw ;
237  }
238 
239  // Send a TCP shutdown
240  beast::error_code ec;
241  stream.socket().shutdown(tcp::socket::shutdown_send, ec);
242 
243  // At this point the connection is closed gracefully
244  // we ignore the error because the client might have
245  // dropped the connection already.
246  }

250  // Accepts incoming connections and launches the sessions
251  inline net::awaitable<void> do_listen(
252  tcp::endpoint endpoint,
253  std::shared_ptr<std::string const> doc_root)
254  {
255  // Open the acceptor
256  auto acceptor = net::use_awaitable.as_default_on(tcp::acceptor(co_await net::this_coro::executor));
257  acceptor.open(endpoint.protocol());
258 
259  // Allow address reuse
260  acceptor.set_option(net::socket_base::reuse_address(true));
261 
262  // Bind to the server address
263  acceptor.bind(endpoint);
264 
265  // Start listening for connections
266  acceptor.listen(net::socket_base::max_listen_connections);
267 
268  for(;;)
269  boost::asio::co_spawn(
270  acceptor.get_executor(),
271  do_session(tcp_stream(co_await acceptor.async_accept()), doc_root),
272  [](std::exception_ptr e)
273  {
274  if (e)
275  try
276  {
277  std::rethrow_exception(e);
278  }
279  catch (std::exception &e) {
280  std::cerr << "Error in session: " << e.what() << "\n";
281  }
282  });

286  inline int run(int argc, char* argv[])
287  {
288  // Check command line arguments.
289  if (argc != 5)
290  {
291  std::cerr <<
292  "Usage: http-server-awaitable <address> <port> <doc_root> <threads>\n" <<
293  "Example:\n" <<

295  return EXIT_FAILURE;
296  }
297  auto const address = net::ip::make_address(argv[1]);
298  auto const port = static_cast<unsigned short>(std::atoi(argv[2]));
299  auto const doc_root = std::make_shared<std::string>(argv[3]);
300  auto const threads = std::max<int>(1, std::atoi(argv[4]));
301 
302  // The io_context is required for all I/O
303  net::io_context ioc{threads};
304 
305  // Spawn a listening port
306  boost::asio::co_spawn(ioc,
307  do_listen(tcp::endpoint{address, port}, doc_root),
308  [](std::exception_ptr e)
309  {
310  if (e)
311  try
312  {
313  std::rethrow_exception(e);
314  }
315  catch(std::exception & e)
316  {
317  std::cerr << "Error in acceptor: " << e.what() << "\n";
318  }
319  });
320 
321  // Run the I/O service on the requested number of threads
322  std::vector<std::thread> v;
323  v.reserve(threads - 1);
324  for(auto i = threads - 1; i > 0; --i)
325  v.emplace_back(
326  [&ioc]
327  {
328  ioc.run();
329  });
330  ioc.run();
331 
332  return EXIT_SUCCESS;
333  }
